B. Analisis Kebutuhan
• File audio disimpan di res/raw/ (tidak memerlukan akses storage eksternal).
• Pemutaran dikendalikan dengan tombol Play, Pause, dan Stop.
• SeekBar menampilkan dan mengontrol posisi pemutaran secara interaktif.
• Aplikasi harus mengosongkan resource (release) saat Activity dihancurkan.

D. Implementasi — Activity (PlayerActivity.kt)
package com.example.audioplayer
import android.media.MediaPlayer
import android.os.Bundle
import android.os.Handler
import android.os.Looper
import android.widget.*
import androidx.appcompat.app.AppCompatActivity
class PlayerActivity : AppCompatActivity() {
private var mediaPlayer: MediaPlayer? = null
private val handler = Handler(Looper.getMainLooper())
private lateinit var seekBar: SeekBar
private lateinit var tvCurrent: TextView
private lateinit var tvDurasi: TextView
override fun onCreate(savedInstanceState: Bundle?) {
super.onCreate(savedInstanceState)
setContentView(R.layout.activity_player)
seekBar = findViewById(R.id.seekBar)
tvCurrent = findViewById(R.id.tvCurrent)
tvDurasi = findViewById(R.id.tvDurasi)
// Inisialisasi MediaPlayer dari resource res/raw/apple.mp3
mediaPlayer = MediaPlayer.create(this, R.raw.apple).apply {
setOnPreparedListener {
seekBar.max = duration
tvDurasi.text = formatWaktu(duration)
}
setOnCompletionListener { resetPlayer() }
}
// Tombol Play
findViewById<Button>(R.id.btnPlay).setOnClickListener {
mediaPlayer?.let { mp ->
if (!mp.isPlaying) {
mp.start()
updateSeekBar() // Mulai update seekbar
}
}
}
// Tombol Pause
findViewById<Button>(R.id.btnPause).setOnClickListener {
mediaPlayer?.takeIf { it.isPlaying }?.pause()
}
// Tombol Stop
findViewById<Button>(R.id.btnStop).setOnClickListener { resetPlayer() }
// SeekBar dragged by user
seekBar.setOnSeekBarChangeListener(object :
SeekBar.OnSeekBarChangeListener {

override fun onProgressChanged(sb: SeekBar, progress: Int, fromUser:
Boolean) {
if (fromUser) {
mediaPlayer?.seekTo(progress)
tvCurrent.text = formatWaktu(progress)
}
}
override fun onStartTrackingTouch(sb: SeekBar) {}
override fun onStopTrackingTouch(sb: SeekBar) {}
})
}
// Perbarui posisi SeekBar setiap 500ms
private fun updateSeekBar() {
handler.postDelayed({
mediaPlayer?.let { mp ->
if (mp.isPlaying) {
seekBar.progress = mp.currentPosition
tvCurrent.text = formatWaktu(mp.currentPosition)
updateSeekBar() // Rekursif
}
}
}, 500)
}
// Reset ke posisi awal
private fun resetPlayer() {
mediaPlayer?.apply { if (isPlaying) stop(); reset() }
handler.removeCallbacksAndMessages(null)
seekBar.progress = 0
tvCurrent.text = "0:00"
}
// Konversi ms ke format m:ss
private fun formatWaktu(ms: Int): String {
val detik = (ms / 1000) % 60
val menit = ms / 60000
return "%d:%02d".format(menit, detik)
}
// Bebaskan resource saat Activity dihancurkan
override fun onDestroy() {
super.onDestroy()
handler.removeCallbacksAndMessages(null)
mediaPlayer?.release()
mediaPlayer = null
}
}
E. Penjelasan Kode
Kode di atas mengimplementasikan pemutar musik sederhana dengan beberapa poin penting:
• MediaPlayer.create() digunakan karena file audio tersedia secara lokal di res/raw/,
sehingga tidak perlu memanggil prepare() secara manual.
• Handler.postDelayed() dengan rekursi digunakan untuk memperbarui posisi
SeekBar setiap 500 milidetik tanpa memblokir thread utama.
• Metode release() dipanggil di onDestroy() untuk memastikan tidak ada memory
leak akibat resource audio yang tidak dilepaskan.
